# =============================================================================
# QwenSwarm — requirements.txt
# =============================================================================
# Two distinct runtime environments exist in this project:
#
#   1. MODAL CONTAINER (modal_server.py)
#      These packages are baked into the remote GPU image at build-time via
#      modal's Image.pip_install(). They do NOT need to be listed here for the
#      server to work, but are documented so developers can reproduce the image
#      or run SGLang locally for debugging.
#
#   2. LOCAL ORCHESTRATOR (orchestrator.py)
#      These packages run on the developer's laptop / CI machine and drive the
#      swarm logic over HTTPS against the Modal endpoint.
#
# Install locally with:  pip install -r requirements.txt
# =============================================================================

# ---------------------------------------------------------------------------
# ORCHESTRATOR — local runtime dependencies
# ---------------------------------------------------------------------------

# Async OpenAI client — the single wire between the orchestrator and the
# SGLang-backed Modal endpoint.  We use AsyncOpenAI so every agent call is
# a non-blocking coroutine, allowing the event-loop to pipeline work.
openai>=1.35.0

# Pydantic v2 — used for two purposes:
#   a) Defining strongly-typed output schemas (Plan, Execution, Review).
#   b) Serialising those schemas to JSON Schema which is forwarded to SGLang
#      as the `response_format` body. SGLang's XGrammar layer converts the
#      JSON Schema into a finite-state automaton and applies GPU-side logit
#      masking — no CPU JSON parsing is involved in token generation.
pydantic>=2.7.0

# Rich — pretty async-aware console output for the hackathon demo.
rich>=13.7.0

# python-dotenv — load MODAL_ENDPOINT_URL and other secrets from .env
python-dotenv>=1.0.0

# httpx is pulled in transitively by openai>=1.x but we pin it explicitly
# to avoid resolver conflicts with modal's own httpx usage.
